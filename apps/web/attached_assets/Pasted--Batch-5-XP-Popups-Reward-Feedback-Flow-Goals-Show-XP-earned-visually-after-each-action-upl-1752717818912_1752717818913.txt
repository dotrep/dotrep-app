✅ 1. Create ToastManager.jsx Component
Reusable notification/toast system.

🔧 components/ToastManager.jsx
jsx
Copy
Edit
import React, { useEffect, useState } from 'react';

const ToastManager = ({ toasts }) => {
  return (
    <div className="toast-container">
      {toasts.map((toast, index) => (
        <div key={index} className={`toast ${toast.type}`}>
          {toast.icon && <span className="toast-icon">{toast.icon}</span>}
          <span className="toast-message">{toast.message}</span>
        </div>
      ))}
    </div>
  );
};

export default ToastManager;
🔧 In global state:
Add a new array activeToasts and a function to push:

js
Copy
Edit
function showToast({ message, icon = '', type = 'xp', duration = 2500 }) {
  activeToasts.push({ message, icon, type });
  setTimeout(() => removeToast(), duration);
}
